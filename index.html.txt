<script>
    // Constants for TN grid and Gas CO2
    const GRID_CO2_INTENSITY = 0.29; // kg per kWh
    const GAS_CO2_INTENSITY = 8.89;  // kg per gallon

    function calculate() {
        // Get input values
        const miles = parseFloat(document.getElementById('annualMiles').value) || 0;
        const gasPrice = parseFloat(document.getElementById('gasPrice').value) || 0;
        const mpg = parseFloat(document.getElementById('iceMpg').value) || 1; // prevent div by zero
        const rate = parseFloat(document.getElementById('elecRate').value) || 0;
        const evEff = parseFloat(document.getElementById('evEff').value) || 1;

        // Cost Calculations
        const gasCost = (miles / mpg) * gasPrice;
        const evCost = (miles / evEff) * rate;
        const savings = gasCost - evCost;

        // CO2 Calculations
        const gasCo2 = (miles / mpg) * GAS_CO2_INTENSITY;
        const evCo2 = (miles / evEff) * GRID_CO2_INTENSITY;
        const co2Reduction = gasCo2 - evCo2;

        // Update UI
        document.getElementById('resGasCost').innerText = gasCost.toLocaleString(undefined, {minimumFractionDigits: 2, maximumFractionDigits: 2});
        document.getElementById('resEvCost').innerText = evCost.toLocaleString(undefined, {minimumFractionDigits: 2, maximumFractionDigits: 2});
        document.getElementById('resSavings').innerText = savings.toLocaleString(undefined, {minimumFractionDigits: 2, maximumFractionDigits: 2});
        
        document.getElementById('resGasCo2').innerText = gasCo2.toLocaleString(undefined, {maximumFractionDigits: 0});
        document.getElementById('resEvCo2').innerText = evCo2.toLocaleString(undefined, {maximumFractionDigits: 0});
        document.getElementById('resCo2Reduction').innerText = co2Reduction.toLocaleString(undefined, {maximumFractionDigits: 0});
    }

    // Run once on load
    calculate();
</script>
